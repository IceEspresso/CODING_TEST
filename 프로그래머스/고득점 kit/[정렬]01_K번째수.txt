import java.util.*;

class Solution {
    public int[] solution(int[] array, int[][] commands) {
        int[] answer = new int [commands.length];
        for(int i=0;i<commands.length;i++){
            int a, b, c;
            a = commands[i][0];
            b = commands[i][1];
            c = commands[i][2];
            int[] tmp = Arrays.copyOfRange(array, a-1, b);
            Arrays.sort(tmp);
            answer[i] = tmp[c-1];
        }
        
        return answer;
    }
}
